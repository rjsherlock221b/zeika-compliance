Set Up Database Tables
Copy this SQL into your Supabase SQL Editor → New Query and click Run.
-- ZEIKA Schema v2 — paste into Supabase SQL Editor create extension if not exists "uuid-ossp"; create table if not exists app_users(id uuid primary key default uuid_generate_v4(),name text not null,email text unique not null,phone text,password_hash text,role_name text default 'Staff',branch_name text default 'Main',status text default 'active',reset_token text,reset_expires timestamptz,last_login timestamptz,created_at timestamptz default now()); create table if not exists roles(id uuid primary key default uuid_generate_v4(),name text unique not null,permissions jsonb default '{}',is_custom boolean default false,created_at timestamptz default now()); create table if not exists branches(id uuid primary key default uuid_generate_v4(),name text unique not null,address text,status text default 'active',created_at timestamptz default now()); create table if not exists transactions(id uuid primary key default uuid_generate_v4(),type text not null,cat text not null,description text not null,amount numeric(12,2) not null,pay text default 'Cash',date date not null,notes text,recurring text default 'none',subcat text,receipt_url text,submitted_by text,approved_by text,branch_name text,created_at timestamptz default now()); create table if not exists daily_sales(id uuid primary key default uuid_generate_v4(),sale_date date not null,branch_name text,items jsonb default '[]',total_income numeric(12,2) default 0,total_expense numeric(12,2) default 0,net numeric(12,2) default 0,status text default 'pending',submitted_by text,approved_by text,notes text,created_at timestamptz default now()); create table if not exists budgets(id uuid primary key default uuid_generate_v4(),month text not null,cat text not null,amount numeric(12,2) default 0,unique(month,cat)); create table if not exists categories(id uuid primary key default uuid_generate_v4(),type text not null,name text not null,subcats jsonb default '[]',color text,created_at timestamptz default now(),unique(type,name)); create table if not exists invoices(id uuid primary key default uuid_generate_v4(),invoice_no text unique not null,client_name text not null,client_email text,client_addr text,issue_date date not null,due_date date,items jsonb default '[]',subtotal numeric(12,2) default 0,vat_amount numeric(12,2) default 0,total numeric(12,2) default 0,status text default 'draft',notes text,created_by text,created_at timestamptz default now()); create table if not exists inventory(id uuid primary key default uuid_generate_v4(),name text not null,category text,sku text,qty numeric(12,2) default 0,unit text default 'units',cost_price numeric(12,2) default 0,sell_price numeric(12,2) default 0,min_stock numeric(12,2) default 0,supplier text,notes text,last_updated timestamptz default now(),created_at timestamptz default now());
